✅ TAREFA 2: CRIAR stores/auth.ts
Arquivo: stores/auth.ts
O que fazer:
•	Criar store unificada com Pinia
•	Gerenciar autenticação do site principal E da agência
•	Implementar getters para currentUser e currentToken (com fallback)
•	Implementar actions: setUser, setAgenciaUser, logout, hydrate
•	Integrar com localStorage
Código a implementar:
import { defineStore } from 'pinia';

interface User {
  id: string;<br/>
  name: string;<br/>
  email: string;<br/>
  admin?: boolean;
}

export const useAuthStore = defineStore('auth', {
  state: () => ({<br/>
    user: null as User | null,<br/>
    token: null as string | null,<br/>
    agenciaUser: null as User | null,<br/>
    agenciaToken: null as string | null,
  }),

  getters: {<br/>
    isAuthenticated: (state) => !!state.token,<br/>
    isAgenciaAuthenticated: (state) => !!state.agenciaToken,<br/>
    currentUser: (state) => state.agenciaUser || state.user,<br/>
    currentToken: (state) => state.agenciaToken || state.token,
  },

  actions: {<br/>
    setUser(user: User, token: string) {
      this.user = user;
      this.token = token;
      localStorage.setItem('auth_token', token);
      localStorage.setItem('user', JSON.stringify(user));
    },

    setAgenciaUser(user: User, token: string) {
      this.agenciaUser = user;
      this.agenciaToken = token;
      localStorage.setItem('agencia_token', token);
      localStorage.setItem('agencia_user', JSON.stringify(user));
    },

    logout() {
      this.user = null;
      this.token = null;
      this.agenciaUser = null;
      this.agenciaToken = null;
      localStorage.removeItem('auth_token');
      localStorage.removeItem('user');
      localStorage.removeItem('agencia_token');
      localStorage.removeItem('agencia_user');
    },

    hydrate() {
      const token = localStorage.getItem('auth_token');
      const user = localStorage.getItem('user');
      const agenciaToken = localStorage.getItem('agencia_token');
      const agenciaUser = localStorage.getItem('agencia_user');

      if (token && user) {
        try {
          this.token = token;
          this.user = JSON.parse(user);
        } catch (e) {
          console.error('Erro ao parsear user:', e);
          this.logout();
        }
      }

      if (agenciaToken && agenciaUser) {
        try {
          this.agenciaToken = agenciaToken;
          this.agenciaUser = JSON.parse(agenciaUser);
        } catch (e) {
          console.error('Erro ao parsear agenciaUser:', e);
          this.logout();
        }
      }
    },
  },
});

- Verificar se há token (agência ou usuário principal)
- Proteger rotas `/agencia/painel/*`
- Verificar permissão de admin para `/agencia/painel/admin/*`
- Redirecionar inteligentemente

**Código a implementar:**
```typescript
import { defineNuxtRouteMiddleware, navigateTo } from '#app';
import { useAuthStore } from '~/stores/auth';

export default defineNuxtRouteMiddleware((to, from) => {
  const authStore = useAuthStore();

  if (to.path.startsWith('/agencia/painel')) {
    if (!authStore.currentToken) {
      return navigateTo('/agencia/login');
    }

    if (to.path.startsWith('/agencia/painel/admin')) {
      const isAdmin = authStore.currentUser?.admin === true;
      if (!isAdmin) {
        return navigateTo('/agencia/painel');
      }
    }
  }
});

TAREFA 5: IMPLEMENTAR VALIDAÇÃO JWT NO SERVIDOR
Arquivo: server/middleware/verify-jwt.ts
O que fazer:
•	Verificar assinatura JWT no servidor
•	Armazenar usuário no contexto do evento
•	Tratar erro 401 (token inválido ou expirado)
Código a implementar:
import jwt from 'jsonwebtoken';
import { defineEventHandler, getCookie, createError, useRuntimeConfig } from 'h3';

declare module 'h3' {
  interface H3EventContext {
    user?: { id: string; email: string; admin?: boolean; };
  }
}

export default defineEventHandler(async (event) => {
  const token = getCookie(event, 'auth_token') || getCookie(event, 'agencia_token');
  
  if (!token) {
    return;
  }

  try {
    const config = useRuntimeConfig();
    const jwtSecret = config.jwtSecret || process.env.NUXT_JWT_SECRET;

    if (!jwtSecret) {
      throw createError({
        statusCode: 500,<br/>
        statusMessage: 'Configuração de segurança inválida.'
      });
    }

    const decoded = jwt.verify(token, jwtSecret) as { id: string; email: string; admin?: boolean; };
    event.context.user = decoded;

  } catch (error) {
    console.warn('Token JWT inválido ou expirado:', error);
    throw createError({
      statusCode: 401,<br/>
      statusMessage: 'Não autorizado: Token inválido ou expirado.'
    });
  }
});
