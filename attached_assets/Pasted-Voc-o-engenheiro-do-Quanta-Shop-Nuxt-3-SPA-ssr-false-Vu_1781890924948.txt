 * useCupons — fonte de cupons/promoções da Quanta Shop.

 * Hoje retorna dados FICTÍCIOS no MESMO formato da Awin Promotions API,
 * para validarmos o visual das seções/página antes da integração.
 *
 * Quando o backend expuser o feed real (ex.: GET /api/cupons que encapsula

export interface Cupom {
  /** promotionId na Awin */
  id: string | number;
  /** 'voucher' = tem código | 'promotion' = oferta sem código */
  type: 'voucher' | 'promotion';

  category: string;
  title: string;

   * recebe o dobro. Aqui guardamos o teto (caso assinante). null = sem cashback.

  cashback: string | null;

  code: string | null;

  exclusive?: boolean;

  url: string;

/**
 * Filtra o que a UI pode mostrar:
 * - remove cupons expirados (a Awin manda endDate; nunca mostrar cupom morto);
 * - mantém apenas lojas COM cashback (posicionamento Quanta: "cashback de verdade").
 */
function ativos(list: Cupom[]): Cupom[] {
  const hoje = new Date();
  hoje.setHours(0, 0, 0, 0);
  return list.filter((c) => !!c.cashback && new Date(c.expiresAt) >= hoje);
}

export function useCupons() {
  async function loadCupons() {
    if (loaded.value) return;
    // TODO(integração Awin): trocar por useFetch('/api/cupons')
    cupons.value = ativos(MOCK);
    loaded.value = true;
  }

  /** dias restantes até expirar (para o selo "expira em X dias") */
  function diasRestantes(c: Cupom): number {
    const ms = new Date(c.expiresAt).getTime() - Date.now();
    return Math.max(0, Math.ceil(ms / 86400000));
  }

  return { cupons, loadCupons, diasRestantes, categories: CUPOM_CATEGORIES };
}

const revealed = ref(false);
const copied = ref(false);
const logoFailed = ref(false);

const initials = computed(() =>
  props.cupom.store.name.split(' ').slice(0, 2).map((w) => w[0]).join('').toUpperCase(),
);
const expiraLabel = computed(() => {
  const d = props.diasRestantes;
  if (d <= 0) return 'expira hoje';
  if (d === 1) return 'expira amanhã';
  return `expira em ${d} dias`;
});

function reveal() {
  revealed.value = true;
  // dispara o deeplink de rastreio da Awin (em produção registra a venda)
  if (props.cupom.url && props.cupom.url !== '#') {
    window.open(props.cupom.url, '_blank', 'noopener');
  }
}

async function copy() {
  if (!props.cupom.code) return;
  try { await navigator.clipboard.writeText(props.cupom.code); } catch { /* noop */ }
  copied.value = true;
  setTimeout(() => (copied.value = false), 2000);
}
